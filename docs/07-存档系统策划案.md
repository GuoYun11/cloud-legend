# 存档系统策划案

## 1. 系统概述

存档系统负责所有游戏数据的持久化，采用**本地 localStorage 为主、云端备份为辅**的方案，确保 H5 单机游戏在无网络时也能正常游戏。

## 2. 存储结构

// 本地 localStorage Key 规范
const STORAGE_KEYS = {
  PLAYER_DATA:    'yctl_player',          // 角色基础数据
  INVENTORY:      'yctl_inventory',       // 背包数据
  EQUIPMENT:      'yctl_equipment',       // 穿戴数据
  MAP_PROGRESS:   'yctl_map',             // 地图进度
  ZODIAC:         'yctl_zodiac',          // 生肖数据
  WALLET:         'yctl_wallet',          // 金币/经济数据
  POTIONS:        'yctl_potions',         // 药剂库存与自动补给设置
  MATERIALS:      'yctl_materials',       // 幸运材料库存
  SKILLS:         'yctl_skills',          // 技能等级数据
  COMBAT_STATE:   'yctl_combat',          // 挂机状态（含上次所在地图）
  SETTINGS:       'yctl_settings',        // 游戏设置
  LAST_SAVE:      'yctl_last_save'        // 最后存档时间戳
};

// 完整存档数据
interface FullSaveData {
  version: string;                        // 存档版本，用于迁移
  savedAt: number;                        // 存档时间戳
  player: PlayerData;
  inventory: Inventory;
  equipment: PlayerEquipment;
  mapProgress: MapProgress;
  zodiac: ZodiacProgress;
  wallet: PlayerWallet;
  potions: PotionSaveData;
    materials: MaterialSaveData;
    skills: SkillSaveData;
    combatState: Partial<CombatState>;      // 主要存 mapId / 阵亡态 / 轻量统计
  settings: GameSettings;
}

interface PotionSaveData {
  inventory: Record<string, number>;
  autoSettings: {
    hpEnabled: boolean;
    hpThreshold: number;
    hpPotionId: string;
    mpEnabled: boolean;
    mpThreshold: number;
    mpPotionId: string;
  };
}

interface MaterialSaveData {
  items: Record<string, number>;
}

interface SkillSaveData {
  skillLevels: Record<string, number>;    // skillId -> 当前等级（1~5）
}

## 4. 核心接口

```typescript
class SaveSystem {
  // 存档间隔（毫秒），每30秒自动存一次
  private static AUTO_SAVE_INTERVAL = 30000;

  /** 全量存档 */
  static saveAll(playerId: string): void {
    const data: FullSaveData = {
      version: '1.1.0',
      savedAt: Date.now(),
      player: PlayerSystem.getPlayerData(playerId),
      inventory: EquipmentSystem.getInventory(playerId) as any,
      equipment: EquipmentSystem.getEquipped(playerId),
      mapProgress: MapSystem.getProgress(playerId),
      zodiac: ZodiacSystem.getProgress(playerId),
      wallet: EconomySystem.getWallet(playerId),
      potions: PotionSystem.getData(),
      materials: EquipmentSystem.getMaterials(playerId),
      skills: SkillSystem.getSkillsData(playerId),
      combatState: SaveSystem.loadCombatState(playerId) ?? { mapId: MapSystem.getProgress(playerId).currentMapId },
      settings: SaveSystem.loadSettings()
    };
    localStorage.setItem(STORAGE_KEYS.PLAYER_DATA, JSON.stringify(data.player));
    localStorage.setItem(STORAGE_KEYS.INVENTORY, JSON.stringify(data.inventory));
    localStorage.setItem(STORAGE_KEYS.EQUIPMENT, JSON.stringify(data.equipment));
    localStorage.setItem(STORAGE_KEYS.MAP_PROGRESS, JSON.stringify(data.mapProgress));
    localStorage.setItem(STORAGE_KEYS.ZODIAC, JSON.stringify(data.zodiac));
    localStorage.setItem(STORAGE_KEYS.WALLET, JSON.stringify(data.wallet));
    localStorage.setItem(STORAGE_KEYS.POTIONS, JSON.stringify(data.potions));
    localStorage.setItem(STORAGE_KEYS.MATERIALS, JSON.stringify(data.materials));
    localStorage.setItem(STORAGE_KEYS.SKILLS, JSON.stringify(data.skills));
    localStorage.setItem(STORAGE_KEYS.LAST_SAVE, String(Date.now()));
  }

  /** 读取全量存档（首次启动时调用） */
  static loadAll(): FullSaveData | null {
    try {
      const player = localStorage.getItem(STORAGE_KEYS.PLAYER_DATA);
      if (!player) return null;
      return {
        version: '1.1.0',
        savedAt: Number(localStorage.getItem(STORAGE_KEYS.LAST_SAVE) ?? 0),
        player: JSON.parse(player),
        inventory: JSON.parse(localStorage.getItem(STORAGE_KEYS.INVENTORY) ?? '{}'),
        equipment: JSON.parse(localStorage.getItem(STORAGE_KEYS.EQUIPMENT) ?? '{}'),
        mapProgress: JSON.parse(localStorage.getItem(STORAGE_KEYS.MAP_PROGRESS) ?? '{}'),
        zodiac: JSON.parse(localStorage.getItem(STORAGE_KEYS.ZODIAC) ?? '{}'),
        wallet: JSON.parse(localStorage.getItem(STORAGE_KEYS.WALLET) ?? '{}'),
        potions: JSON.parse(localStorage.getItem(STORAGE_KEYS.POTIONS) ?? '{}'),
        materials: JSON.parse(localStorage.getItem(STORAGE_KEYS.MATERIALS) ?? '{}'),
        skills: JSON.parse(localStorage.getItem(STORAGE_KEYS.SKILLS) ?? '{"skillLevels":{}}'),
        combatState: JSON.parse(localStorage.getItem(STORAGE_KEYS.COMBAT_STATE) ?? '{}'),
        settings: JSON.parse(localStorage.getItem(STORAGE_KEYS.SETTINGS) ?? '{}')
      };
    } catch {
      return null;
    }
  }

  /** 启动自动存档 */
  static startAutoSave(getStateFn: () => FullSaveData): void {
    setInterval(() => SaveSystem.saveAll(getStateFn()), this.AUTO_SAVE_INTERVAL);
  }

  /** 保存战斗状态 */
  static saveCombatState(state: CombatState): void {
    localStorage.setItem(STORAGE_KEYS.COMBAT_STATE, JSON.stringify(state));
  }

  /** 读取战斗状态 */
  static loadCombatState(playerId: string): Partial<CombatState> | null {
    const raw = localStorage.getItem(STORAGE_KEYS.COMBAT_STATE);
    return raw ? JSON.parse(raw) : null;
  }

  /** 更新离线奖励时间戳 */
  static updateOfflineRewardTime(playerId: string, time: number): void {
    const player = JSON.parse(localStorage.getItem(STORAGE_KEYS.PLAYER_DATA) ?? '{}');
    player.lastOfflineRewardAt = time;
    localStorage.setItem(STORAGE_KEYS.PLAYER_DATA, JSON.stringify(player));
  }

  /** 读取设置 */
  static loadSettings(): GameSettings {
    const raw = localStorage.getItem(STORAGE_KEYS.SETTINGS);
    return raw ? { ...DEFAULT_SETTINGS, ...JSON.parse(raw) } : { ...DEFAULT_SETTINGS };
  }

  /** 保存设置 */
  static saveSettings(settings: Partial<GameSettings>): void {
    const current = SaveSystem.loadSettings();
    localStorage.setItem(STORAGE_KEYS.SETTINGS, JSON.stringify({ ...current, ...settings }));
  }

  /** 清空存档（重新开始） */
  static clearAll(): void {
    Object.values(STORAGE_KEYS).forEach(key => localStorage.removeItem(key));
  }

  /** 导出存档字符串（备份用） */
  static exportSave(playerId: string): string {
    SaveSystem.saveAll(playerId);
    const allData: Record<string, string> = {};
    Object.entries(STORAGE_KEYS).forEach(([, key]) => {
      const val = localStorage.getItem(key);
      if (val) allData[key] = val;
    });
    return btoa(JSON.stringify(allData));
  }

  /** 导入存档字符串 */
  static importSave(saveStr: string): boolean {
    try {
      const allData: Record<string, string> = JSON.parse(atob(saveStr));
      Object.entries(allData).forEach(([key, val]) => localStorage.setItem(key, val));
      return true;
    } catch {
      return false;
    }
  }
}
```

## 5. 存档迁移方案

```typescript
class SaveMigration {
  static migrate(data: FullSaveData): FullSaveData {
    if (data.version === '1.0.0') {
      return {
        ...data,
        version: '1.2.0',
        potions: data.potions ?? { inventory: {}, autoSettings: {} as any },
        materials: data.materials ?? { items: {} },
        skills: data.skills ?? { skillLevels: {} }
      };
    }
    if (data.version === '1.1.0') {
      return {
        ...data,
        version: '1.2.0',
        skills: data.skills ?? { skillLevels: {} }
      };
    }
    if (data.version === '1.2.0') return data;
    return data;
  }
}
```

## 6. AI开发提示

- **P0**：saveAll / loadAll、自动存档、存档初始化
- **P1**：导出/导入存档（玩家换设备时用）
- **P2**：云端备份（可选，接入微信云开发）
- localStorage 单个 key 建议不超过 1MB，背包数据注意精简（只存必要字段）
- 每次启动时调用 `loadAll`，若返回 null 则走新用户创建流程
